alter  function [dbo].[vcca2trd](@project nvarchar(20),@t_custno nvarchar(20),@t_vccano nvarchar(20),@t_trdno nvarchar(20))
returns @tmp table(
		sel int identity(1,1)
		,noa nvarchar(20)
		,datea nvarchar(20)
		,cust nvarchar(50)
		,buyer nvarchar(50)
		,[money] float
		,tax float
		,total float
		,memo nvarchar(max)
		,tax2 float --已立帳稅額
		,tax3 float --未立帳稅額
) as
begin
	insert into @tmp(noa,datea,cust,buyer,[money],tax,total,memo,tax2,tax3)
	select a.noa,a.datea,a.comp,a.buyer,a.[money],a.tax,a.total,a.memo
		,b.tax tax2, isnull(a.tax,0)-isnull(b.tax,0) diff
	from vcca a
	outer apply (select SUM(ISNULL(tax,0)) tax from view_trd where noa!=@t_trdno and vccano=a.noa) b
	where a.custno=@t_custno
	and  isnull(a.tax,0) != isnull(b.tax,0) 
	return 
end
GO
