declare @tmp table(
	gno nvarchar(1),
	noa nvarchar(50),
	ordbno nvarchar(50),
	tggno nvarchar(50),
	tgg nvarchar(50),
	productno nvarchar(100),
	product nvarchar(max),
	spec nvarchar(50),
	unit nvarchar(50),
	omount float,
	mount float,
	price float,
	total float,
	memo nvarchar(max)
)
insert @tmp
select '0',a.noa,a.ordbno,b.tggno,b.tgg,b.productno,b.product,b.spec,b.unit,b.omount,b.mount,b.price
,b.total,b.memo
from ordp a left join ordps b on a.noa=b.noa
where (a.noa=@t_noa or len(@t_noa)=0)

insert @tmp(gno,noa)
select '1',noa
from @tmp
group by noa

select 
dbo.getComma(omount,0)omount
,dbo.getComma(mount,0)mount
,dbo.getComma(price,2)price
,dbo.getComma(total,0)total
,* from @tmp
