	declare @tmprc2 table(
		productno nvarchar(50),
		storeno nvarchar(50),
		mount float,
		weight float
	)
	insert into @tmprc2 (productno,storeno,mount,weight)
	select ISNULL(b.productno,''),ISNULL(b.storeno,'')
	,SUM(ISNULL(case when a.typea='1' then 1 else -1 end*b.mount,0))
	,SUM(ISNULL(case when a.typea='1' then 1 else -1 end*b.[weight],0))
	from view_rc2 a
	left join view_rc2s b on a.accy=b.accy and a.noa=b.noa
	where b.noa is not null
	and ISNULL(b.productno,'') between @t_bproductno and @t_eproductno
	and ISNULL(a.datea,'') between @t_bdate and @t_edate
	and ISNULL(b.storeno,'') between @t_bstoreno and @t_estoreno
