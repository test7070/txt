		where a.ordeno!='' and a.ordeno2 !=''
		
		update @tmp 
		set qhrefa = substring(qhrefa,0,len(qhrefa)-2)+'?left(noa,'+cast(len(noa) as nvarchar)+')=$noa?'+substring(qhrefa,len(qhrefa)-2,len(qhrefa))
		,qhrefb= 'orde_xy?noa=$ordeno?'+ordeaccy
		,qhrefc= case when charindex(',',vccno)>0 then '' else 'vcc_xy?noa=$vccno?'+vccaccy end
		
		insert into @tmp(gno,datea,mount,total,omount,vmount,lengthb,lengthc)
		select '1',datea,sum(mount),sum(total),sum(omount),sum(vmount),SUM(lengthb),SUM(lengthc) from @tmp group by datea
			
		insert into @tmp(gno,datea,mount,total,omount,vmount,lengthb,lengthc)
		select '2',left(datea,6)+'/99',sum(mount),sum(total),sum(omount),sum(vmount),SUM(lengthb),SUM(lengthc) from @tmp where gno='1' group by left(datea,6)
	end
	
	select gno, noa, noq, invono,type, datea, LEFT(datea,6) xdatea, mon, tggno, left(comp,4) comp, productno, xproduct,unit
	,cust,memo
	,dbo.getComma(oprice,-1)oprice 
	,dbo.getComma(omount,-1)omount 
	,ocomp,ordeno,vccno
	,dbo.getComma(vmount,-1)vmount 
	,dbo.getComma(mount,-1) mount 
	,dbo.getComma(weight,-1) weight 
	,dbo.getComma(price,-1) price 
	,dbo.getComma(total,-1) total 
	,row_number()over(partition by datea,left(datea,6) order by datea desc,gno,noa,noq) idno,qhrefa,qhrefb,qhrefc
	,dbo.getComma(lengthb,-1) lengthb 
	,dbo.getComma(lengthc,-1) lengthc 
	,dbo.getComma(radius,-1) radius 
	,xunit
	from @tmp order by datea,gno,noa,noq;
