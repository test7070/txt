	create table #conttype(
		noa nvarchar(20)
	)
	set @string = @t_xstype
	while(1=1)
	begin
		set @n = PATINDEX('%,%',@string)
		if @n=0
		begin
			if LEN(@string)>0
			begin
				insert into #conttype select @string
			end
			break
		end
		insert into #conttype select LEFT(@string,@n-1)	
		set @string = SUBSTRING(@string,@n+1,LEN(@string)-@n)
