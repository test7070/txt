tranvccesh_chk1:--tranvccesh_chk1
SET QUOTED_IDENTIFIER OFF
declare @t_noa nvarchar(50) = [1]
declare @t_noq nvarchar(20)=[2]
declare @accy nvarchar(50)=(select accy from view_tranvcces where noa=@t_noa and no2=@t_noq group by accy)

declare @tmp table(
	msg nvarchar(max)
)

Begin Transaction [Trans_Name]
begin try
	exec("
	update tranvcces"+@accy+"
	set chk1='1'
	where noa='"+@t_noa+"' and no2='"+@t_noq+"'
	")
	insert into @tmp(msg)values('讀單確認!!')
	end try
begin catch
	Rollback Transaction [Trans_Name] -- 復原所有操作所造成的變更
	insert into @tmp(msg)values('讀單錯誤，未有此單號或是此項目!!')
end catch

select * from @tmp
;
