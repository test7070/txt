z_modfixcsp1:--z_modfixcsp1
	SET QUOTED_IDENTIFIER OFF
	declare @t_bdate nvarchar(50)
	declare @t_edate nvarchar(50)
	declare @t_noa nvarchar(50)

	set @t_bdate = case when '#non'=[1] then '' else [1] end
	set @t_edate = case when '#non'=[2] then char(255) else [2] end
	set @t_noa = case when '#non'=[3] then '' else [3] end
	
	declare @tmp table(
		gno nvarchar(1),
		rr int,
		noa nvarchar(100),
		datea nvarchar(10),
		code nvarchar(100),
		detail nvarchar(max),
		nob nvarchar(100),
		product nvarchar(100),
		spec nvarchar(50),
		memo nvarchar(max),
		way2 nvarchar(50),
		worktype nvarchar(100),
		wheel nvarchar(50),
		productno nvarchar(100),
		images nvarchar(max),
		mount float,
		btime nvarchar(50),
		bdate nvarchar(10),
		etime nvarchar(max),
		edate nvarchar(10),
		mech2 nvarchar(50),
		worktype2 nvarchar(10),
		btime2 nvarchar(50),
		etime2 nvarchar(10),
		way3 nvarchar(50),
		bdate2 nvarchar(10),
		edate2 nvarchar(10),
		memo2 nvarchar(max),
		worktype3 nvarchar(50),
		mech3 nvarchar(100),
		btime3 nvarchar(50),
		bdate3 nvarchar(10),
		etime3 nvarchar(50),
		edate3 nvarchar(10),
		memo3 nvarchar(max),
		mech4 nvarchar(100),
		bdate4 nvarchar(30),
		edate4 nvarchar(30),
		btime4 nvarchar(30),
		memo4 nvarchar(max),
		worktype4 nvarchar(max),
		worker4 nvarchar(max),
		frame nvarchar(50),
		mech nvarchar(30),
		way nvarchar(max),
		worker3 nvarchar(max),
		etime4 nvarchar(max),
		way4 nvarchar(max),
		imagememo nvarchar(max)
	)
	
	insert @tmp 
	select '0','',noa,datea,code,detail,nob,product,spec,memo,way2,worktype,wheel,productno
			,images,mount,btime,bdate,etime,edate,mech2,worktype2,btime2,etime2,way3,bdate2
			,edate2,memo2,worktype3,mech3,btime3,bdate3,etime3,edate3,memo3,mech4,bdate4,edate4
			,btime4,memo4,worktype4,worker4,frame,mech,way,worker3,etime4,way4,imagememo
	from modfixcs
	where
	datea between @t_bdate and @t_edate
	and noa=@t_noa
	
	update a
	set rr=rx
	from (select ROW_NUMBER() over(partition by noa order by noa)rx,rr from @tmp)a
	
	
	if ((select count(*) from @tmp)>0)
	begin
	   insert @tmp(gno,noa)
	   select '1',noa from @tmp group by noa
	end
	
	select * from @tmp order by noa,gno
;
