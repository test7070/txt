	declare @t_count int
	set @t_count = 42

	declare @tmp table(
		gno nvarchar(10),
		rr1 int,
		noa1 nvarchar(20),
		product1 nvarchar(max),
		rr2 int,
		noa2 nvarchar(20),
		product2 nvarchar(max)
	)
	
	insert into @tmp(gno,rr1,noa1,product1)
	select '0',* 
	from (select ROW_NUMBER()over(order by noa) rr,noa,product from spec) a
	where floor(rr/@t_count)%2 = 0
	order by rr
	
	update @tmp set rr2=b.rr,noa2=b.noa,product2=b.product
	from @tmp a
	left join (select ROW_NUMBER()over(order by noa) rr,noa,product from spec) b on a.rr1+@t_count=b.rr
	
	select * from @tmp;
