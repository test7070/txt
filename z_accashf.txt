z_accashf01:--z_accashf01
	SET QUOTED_IDENTIFIER OFF 
	declare @cmd nvarchar(max) 
	declare @islock int 
	
	declare @t_accy nvarchar(10) 
	declare @t_mon nvarchar(10)
	------------------------------------------------------------------
	set @t_accy = '[1]'
	set @t_mon = case when '#non'=[2] then '' else [2] end
	----------------------------------------------------------------------------------------------
	
	select a.*
	,b.mon yyy
	,case when a.money1<0 then '('+reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,ABS(isnull(a.money1,0))),1)),4,12))+')'
		else reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,isnull(a.money1,0)),1)),4,12)) end cm1
	,case when a.money2<0 then '('+reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,ABS(isnull(a.money2,0))),1)),4,12))+')'
		else reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,isnull(a.money2,0)),1)),4,12)) end cm2
	from accashfs a 
	left join accashf b on a.noa=b.noa
	where b.mon=@t_mon order by a.sel;
