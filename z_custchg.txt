z_custchg:--z_custchg
declare @t_bdate nvarchar(10)
declare @t_edate nvarchar(10)
declare @t_bcustno nvarchar(20)
declare @t_ecustno nvarchar(20)
set @t_bdate = case when '#non'=[1] then '' else [1] end
set @t_edate = case when '#non'=[2] then CHAR(255) else [2] end
set @t_bcustno = case when '#non'=[3] then '' else [3] end
set @t_ecustno = case when '#non'=[4] then CHAR(255) else [4] end
declare @result table(
		gno  nvarchar(1),
		noa nvarchar(20),
		datea nvarchar(10),
		custno nvarchar(20),
		comp nvarchar(50),
		item nvarchar(50),
		plus float,
		minus float,
		memo nvarchar(max)
)
insert into @result
select '0' gno,a.noa,a.datea,a.custno,case when len(b.nick)>0 then b.nick else left(a.comp,4) end,a.plusitem+a.minusitem,a.plusmoney,a.minusmoney,a.memo
from custchg a
left join cust b on b.noa = a.custno
where (a.datea between @t_bdate and @t_edate) and
(a.custno between @t_bcustno and @t_ecustno)

insert into @result
select '1' gno,'','',custno,'','',SUM(plus),SUM(minus),''
from @result
group by custno

insert into @result
select '2' gno,'','',MAX(custno),'','',SUM(plus),SUM(minus),''
from @result
where not(gno = 1)

select gno,noa,datea,custno,comp,item,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,plus),1)),4,12)) plus,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,minus),1)),4,12)) minus,
memo
from @result
order by custno,gno,datea;

declare @t_bdate nvarchar(10)
declare @t_edate nvarchar(10)
declare @t_bcustno nvarchar(20)
declare @t_ecustno nvarchar(20)
set @t_bdate = case when '#non'=[1] then '' else [1] end
set @t_edate = case when '#non'=[2] then CHAR(255) else [2] end
set @t_bcustno = case when '#non'=[3] then '' else [3] end
set @t_ecustno = case when '#non'=[4] then CHAR(255) else [4] end
declare @cmb table(
		noa nvarchar(20),
		datea nvarchar(10),
		custno nvarchar(20),
		comp nvarchar(50),
		item nvarchar(50),
		plus float,
		minus float,
		memo nvarchar(max)
)
insert into @cmb
select a.noa,a.datea,a.custno,case when len(b.nick)>0 then b.nick else left(a.comp,4) end,a.plusitem+a.minusitem,a.plusmoney,a.minusmoney,a.memo
from custchg a
left join cust b on b.noa = a.custno
where (a.datea between @t_bdate and @t_edate) and
(a.custno between @t_bcustno and @t_ecustno)

declare @tmp table(
		gno  nvarchar(1),
		custno nvarchar(20),
		comp nvarchar(50),
		plus float,
		minus float
)
insert into @tmp
select '0' gno,custno,left(comp,4),SUM(plus),SUM(minus)
from @cmb
group by custno,comp
declare @result table(
		gno nvarchar(1),
		a1 nvarchar(20),
		b1 float,
		c1 float,
		a2 nvarchar(20),
		b2 float,
		c2 float
)
insert into @result
	select
		'0' gno,S1.a1 a1,S1.b1 b1,S1.c1 c1,S2.a2 a2,S2.b2 b2,S2.c2 c2
	from
		(select
			ROW_NUMBER()over(order by R1.comp)as zz,R1.comp a1,R1.plus b1,R1.minus c1
		from
			(select ROW_NUMBER()over(order by comp)as zz, comp, plus,minus from @tmp )as R1
		where R1.zz%2=1) as S1
	left join
		(select
			ROW_NUMBER()over(order by R1.comp)as zz,R1.comp a2,R1.plus b2,R1.minus c2
		from
			(select ROW_NUMBER()over(order by comp)as  zz, comp, plus,minus from @tmp )as R1
		where R1.zz%2=0) as S2
	on s1.zz=s2.zz
	order by S1.zz
	
insert into @result
select 	'1' gno,'',0,0,'',sum(b1)+sum(b2),sum(c1)+sum(c2)
from @result

select gno,a1,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,b1),1)),4,12)) b1,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,c1),1)),4,12)) c1,
a2,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,b2),1)),4,12)) b2,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,c2),1)),4,12)) c2
from @result;
